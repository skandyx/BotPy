# --- CORE SETTINGS ---
# Mode de l'environnement ('development' ou 'production').

NODE_ENV=production

# Le port sur lequel le serveur backend écoutera.

# Le mot de passe que vous utiliserez pour vous connecter au tableau de bord.
# Choisissez un mot de passe fort et unique !
APP_PASSWORD=votre_mot_de_passe_secret

# --- TRADING PARAMETERS ---
# Le solde de départ pour le mode de trading Virtuel.
INITIAL_VIRTUAL_BALANCE=10000

# Le nombre maximum de positions que le bot peut ouvrir simultanément.
MAX_OPEN_POSITIONS=5

# Le pourcentage du solde total à utiliser pour chaque nouvelle position (ex: 2.0 pour 2%).

# Le pourcentage de profit auquel une position est automatiquement fermée.
TAKE_PROFIT_PCT=4.0

# Le pourcentage de perte auquel une position est automatiquement fermée.

# Activer/désactiver le Trailing Stop Loss (true/false).

# Le pourcentage sous le prix le plus élevé pour le Trailing Stop Loss (ex: 1.5 pour 1.5%).
TRAILING_STOP_LOSS_PCT=1.5

# Pourcentage de slippage pour simuler les conditions de marché réelles.
SLIPPAGE_PCT=0.05

# --- STRATEGY & SCANNER FILTERS ---
# Le volume de trading minimum sur 24h (en USD) qu'une paire doit avoir pour être scannée.

# La volatilité de prix minimum qu'une paire doit avoir pour être considérée.
MIN_VOLATILITY_PCT=0.5

# À quelle fréquence (en secondes) le bot doit-il rafraîchir la liste des paires depuis CoinGecko.

# Une liste de paires à ignorer, séparées par des virgules (ex: USDCUSDT,FDUSDUSDT).

# Filtres de stratégie avancés (true/false).
USE_VOLUME_CONFIRMATION=true
USE_MULTI_TIMEFRAME_CONFIRMATION=true

# Après une transaction perdante sur un symbole, combien d'heures le bot doit-il attendre avant de pouvoir trader à nouveau ce symbole.
LOSS_COOLDOWN_HOURS=4
